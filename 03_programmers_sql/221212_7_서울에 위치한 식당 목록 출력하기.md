## MySQL

```sql
SELECT REST_ID, REST_NAME, FOOD_TYPE, FAVORITES, ADDRESS, SCORE
  FROM REST_INFO RI
  JOIN (SELECT REST_ID, FORMAT(AVG(REVIEW_SCORE), 2) SCORE
          FROM REST_REVIEW
         GROUP BY REST_ID) RR USING (REST_ID)
 WHERE LEFT(ADDRESS, 2) = '서울'
 ORDER BY SCORE DESC, FAVORITES DESC

SELECT REST_ID, REST_NAME, FOOD_TYPE, FAVORITES, ADDRESS, ROUND(AVG(REVIEW_SCORE), 2) SCORE
  FROM REST_INFO RI
  JOIN REST_REVIEW USING (REST_ID)
 WHERE LEFT(ADDRESS, 2) = '서울'
 GROUP BY REST_ID
 ORDER BY SCORE DESC, FAVORITES DESC
```

## Oracle

```sql
WITH LAST AS
(
    SELECT REST_ID, ROUND(AVG(REVIEW_SCORE), 2) AS SCORE
      FROM REST_REVIEW
     GROUP BY REST_ID
)
SELECT A.REST_ID, REST_NAME, FOOD_TYPE, FAVORITES, ADDRESS, SCORE
  FROM REST_INFO A, LAST B
 WHERE A.REST_ID = B.REST_ID
   AND ADDRESS LIKE '서울%'
 ORDER BY SCORE DESC, FAVORITES DESC
```
