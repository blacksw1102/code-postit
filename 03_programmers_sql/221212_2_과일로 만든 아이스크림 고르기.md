## MySQL

```sql
SELECT II.FLAVOR
  FROM ICECREAM_INFO II
 INNER JOIN FIRST_HALF JFH ON JFH.FLAVOR = II.FLAVOR
 WHERE JFH.TOTAL_ORDER > 3000
   AND II.INGREDIENT_TYPE = 'fruit_based'
 ORDER BY JFH.TOTAL_ORDER DESC

 SELECT II.FLAVOR
   FROM ICECREAM_INFO II, FIRST_HALF JFH
  WHERE JFH.FLAVOR = II.FLAVOR
    AND JFH.TOTAL_ORDER > 3000
    AND II.INGREDIENT_TYPE = 'fruit_based'
  ORDER BY JFH.TOTAL_ORDER DESC

 SELECT A.FLAVOR
   FROM (SELECT *
           FROM FIRST_HALF
          WHERE TOTAL_ORDER > 3000) A
       ,(SELECT *
           FROM ICECREAM_INFO
          WHERE INGREDIENT_TYPE = 'fruit_based') B
  WHERE A.FLAVOR = B.FLAVOR
  ORDER BY A.TOTAL_ORDER DESC

 SELECT FH.FLAVOR
   FROM FIRST_HALF AS FH
   JOIN ICECREAM_INFO AS II USING (FLAVOR)
  WHERE TOTAL_ORDER > 3000
    AND INGREDIENT_TYPE = 'fruit_based'
  ORDER BY TOTAL_ORDER DESC

 SELECT FH.FLAVOR
   FROM FIRST_HALF AS FH
  WHERE TOTAL_ORDER > 3000
    AND FLAVOR IN (SELECT FLAVOR
                     FROM ICECREAM_INFO
                    WHERE INGREDIENT_TYPE =  'fruit_based')
  ORDER BY TOTAL_ORDER DESC
```
