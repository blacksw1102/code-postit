## MySQL

```sql
SELECT HOUR(DATETIME) AS `HOUR`
     , COUNT(DATETIME) AS `COUNT`
  FROM ANIMAL_OUTS
 WHERE HOUR(DATETIME) BETWEEN '09' AND '19'
 GROUP BY `HOUR`
 ORDER BY `HOUR` ASC ;

SELECT EXTRACT(HOUR FROM DATETIME) AS `HOUR`
     , COUNT(ANIMAL_ID) AS `COUNT`
  FROM ANIMAL_OUTS
 GROUP BY `HOUR`
HAVING `HOUR` >= 9 AND `HOUR` < 20
 ORDER BY `HOUR`;
```
